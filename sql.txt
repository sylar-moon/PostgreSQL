create table stores
(
	id serial not null
		constraint stores_pk
			primary key,
	address varchar(40) not null,
	count_goods int
);

create table goods
             (
             	id_goods int
             		constraint goods_pk
             			primary key,
             	name varchar(50),
             	type varchar(50),
             	brand varchar(50)
             )
			 
create table types
(
    id        serial not null
        constraint type_pk
            primary key,
    name_type integer
);

alter table type
    owner to sylar;

create table brands
(
	id serial
		constraint brands_pk
			primary key
);
